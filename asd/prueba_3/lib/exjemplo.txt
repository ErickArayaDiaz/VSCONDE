import 'package:flutter/material.dart';
import 'package:math_expressions/math_expressions.dart';

void main() => runApp(const MyApp());

class MyApp extends StatelessWidget {
  const MyApp({super.key});
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Calculadora Flutter',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: const CalculatorPage(),
    );
  }
}

class CalculatorPage extends StatefulWidget {
  const CalculatorPage({super.key});
  @override
  State<CalculatorPage> createState() => _CalculatorPageState();
}

class _CalculatorPageState extends State<CalculatorPage> {
  String expression = '';
  String result = '0';

  final List<String> buttons = [
    'C','DEL','%','/',
    '7','8','9','*',
    '4','5','6','-',
    '1','2','3','+',
    '0','.','±','='
  ];

  void onButtonPressed(String label) {
    setState(() {
      if (label == 'C') {
        expression = '';
        result = '0';
      } else if (label == 'DEL') {
        if (expression.isNotEmpty) {
          expression = expression.substring(0, expression.length - 1);
        }
      } else if (label == '=') {
        _calculateResult();
      } else if (label == '±') {
        // toggle sign of whole expression simple approach
        if (expression.startsWith('-')) expression = expression.substring(1);
        else expression = '-$expression';
      } else {
        // evita operadores repetidos básicos: si la última es operador y el nuevo también, reemplaza
        if (_isOperator(label) && expression.isNotEmpty && _isOperator(expression.characters.last)) {
          expression = expression.substring(0, expression.length - 1) + label;
        } else {
          expression += label;
        }
      }
    });
  }

  bool _isOperator(String s) {
    const ops = ['+','-','*','/','%'];
    return ops.contains(s);
  }

  void _calculateResult() {
    try {
      // Reemplaza símbolos si usas otros (aquí usamos * y /)
      String exp = expression.replaceAll('×', '*').replaceAll('÷', '/');
      Parser p = Parser();
      Expression ex = p.parse(exp);
      ContextModel cm = ContextModel();
      double eval = ex.evaluate(EvaluationType.REAL, cm);
      // Simplificar resultado (si entero, sin .0)
      result = eval % 1 == 0 ? eval.toInt().toString() : eval.toString();
    } catch (e) {
      result = 'Error';
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Calculadora')),
      body: Column(
        children: [
          Expanded(
            flex: 2,
            child: Container(
              padding: const EdgeInsets.all(16),
              alignment: Alignment.bottomRight,
              color: Colors.black12,
              child: Column(
                mainAxisAlignment: MainAxisAlignment.end,
                crossAxisAlignment: CrossAxisAlignment.end,
                children: [
                  Text(expression, style: const TextStyle(fontSize: 28)),
                  const SizedBox(height: 8),
                  Text(result, style: const TextStyle(fontSize: 48, fontWeight: FontWeight.bold)),
                ],
              ),
            ),
          ),
          Expanded(
            flex: 5,
            child: GridView.builder(
              padding: const EdgeInsets.all(12),
              gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
                crossAxisCount: 4,
                childAspectRatio: 1.1,
                crossAxisSpacing: 12,
                mainAxisSpacing: 12,
              ),
              itemCount: buttons.length,
              itemBuilder: (context, index) {
                final label = buttons[index];
                // Colores según tipo de botón
                Color bg;
                Color fg = Colors.white;
                if (label == 'C') bg = Colors.red;
                else if (label == 'DEL') bg = Colors.orange;
                else if (label == '=') bg = Colors.green;
                else if (_isOperator(label)) bg = Colors.blueGrey;
                else bg = Colors.blue;

                return ElevatedButton(
                  onPressed: () => onButtonPressed(label),
                  style: ElevatedButton.styleFrom(
                    backgroundColor: bg,
                    shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(12)),
                    padding: const EdgeInsets.all(14),
                  ),
                  child: Text(label, style: TextStyle(fontSize: 24, color: fg)),
                );
              },
            ),
          ),
        ],
      ),
    );
  }
}
