import 'package:flutter/material.dart';
import 'package:scribble/scribble.dart';

class ScribblePage extends StatefulWidget {
  @override
  State<ScribblePage> createState() => _ScribblePageState();
}

class _ScribblePageState extends State<ScribblePage> {
  final notifier = ScribbleNotifier();

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text("Scribble Example")),
      body: Column(
        children: [
          Expanded(
            child: Scribble(
              notifier: notifier,
              drawPen: true, // activa lápiz
            ),
          ),
          Row(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              IconButton(
                icon: const Icon(Icons.undo),
                onPressed: notifier.canUndo ? notifier.undo : null,
              ),
              IconButton(
                icon: const Icon(Icons.redo),
                onPressed: notifier.canRedo ? notifier.redo : null,
              ),
              IconButton(
                icon: const Icon(Icons.clear),
                onPressed: notifier.clear,
              ),
              IconButton(
                icon: const Icon(Icons.color_lens, color: Colors.red),
                onPressed: () {
                  notifier.setColor(Colors.red);
                },
              ),
            ],
          ),
        ],
      ),
    );
  }
}
